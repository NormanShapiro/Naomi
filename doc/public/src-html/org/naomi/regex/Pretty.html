047/** This class governs the display of regular expressions
048(see {@link Pattern#getRegularExpression getRegularExpression} and
049{@link Pattern#setPretty setPretty}), to generate expressions which are more
050human readable&mdash;or, more precisely, less unreadable.
051*/
052public class Pretty
053{
054  private static Pattern whitePattern;
055
056  /** generates a new Pretty with
057  {@link #pretty} == true,
058   {@link #indenter} == 6 spaces
059   and {@link #comments} == true.
060  */
061  public Pretty()
062  {
063     pretty=true;
064     indenter =Utilities.ditto(6," ",null);
065     comments=true;
066  }
067
068 /** generates a new Pretty with
069  {@link #pretty} == prettyArg,
070   {@link #indenter} == 6 spaces
071   and {@link #comments} == true.
072  */
073  public Pretty(boolean prettyArg)
074  {
075     this();
076     pretty=prettyArg;
077  }
078
079  public Pretty(Pretty other)
080  {
081     other.pretty=pretty;
082     other.indenter=indenter;
083     other.comments=comments;
084  }
085
086  /** When true, {@link Pattern#getRegularExpression getRegularExpression}
087  attempts to generate a more human readable expression; when false, all
088  other fields are ignored and
089  {@link Pattern#getRegularExpression getRegularExpression} generates
090  a dense expression. The default is false.
091  */
092  public boolean pretty;
093
094  /** The indentation per level. The default is 6 spaces.*/
095  public CharSequence indenter;
096
097  /** When true, terse comments are generated; when false no comments are
098  generated. The default is true.
099  */
100  public boolean comments;
101
102  private static Pattern getWhitePattern()
103  {
104     if(whitePattern==null)
105        whitePattern=new BuiltInCharClass(CoreBuiltIn.white).setMinAndMaxCount(0,null);
106      return whitePattern;
107 }
108
109  Pretty check()
110  {
111     if(!pretty) return this;
112     if(!getWhitePattern().matches(indenter))
113        throw new BadPrettyException
114           (" indenter=='"+indenter+"' is not all white");
115     return this;
116   }
117
118  /** Thrown when an instance of Pretty has a bad field*/
119  public static class BadPrettyException extends RuntimeException
120  {
121     BadPrettyException(String msg)
122     {
123        super("Bad Pretty: "+ msg);
124     }
125  }
126}
