033package org.naomi.regex;
034
035/** 
036    This Enum provides constant values for the 
037{@link Pattern#setPolicy setPolicy(Policy policy)} method of
038    {@link Pattern} and its subclasses, which controls various behavior and
039    performance aspects of matching for a given pattern.
040<p>
041
042  These behavioral parameters are not interpreted by Naomi but are simply
043  passed along to Java's pattern matching library, whose behavior depends
044  on them in various ways that are beyond the scope of this documentation.
045
046<p>
047
048  Therefore, for an explanation of the effects of these values (when used in
049  the {@link Pattern#setPolicy setPolicy} method) on the behavior and
050  performance aspects of pattern matching, see this discussion of
051
052 <a
053 href="http://www.xyzws.com/Javafaq/what-are-differences-among-greedy-reluctant-and-possessive-quantifiers-in-java-patterns/206">
054 quantifiers</a>.  Additional discussion of these aspects of pattern matching
055 can be found in descriptions of the Perl language's use of regular expressions.
056
057
058
059@see <a href="http://www.xyzws.com/Javafaq/what-are-differences-among-greedy-reluctant-and-possessive-quantifiers-in-java-patterns/206"> quantifiers</a>
060@see <a href="http://search.cpan.org/~jhi/perl-5.8.0/pod/perlre.pod"> Perl
061regular expressions</a>
062
063*/
064public enum Policy
065{
066  /** Is compatible with any any values for min and max count*/
067  greedy("",false)
068  {
069     public Policy checkCounts(Pattern pattern) {return this;}
070  },
071
072  /** Requires that either min count != 1 and/or max count != 1.*/
073  reluctant("?",false),
074
075
076  /** Requires that either min count != 1 and/or max count != 1.*/
077  possessive("+",false),
078
079  /**Requires that min and max count both have their default value, 1.
080  */
081  positiveLookahead("=",true),
082
083  /**Requires that min and max count both have their default value, 1.
084  */
085  negativeLookahead("!",true),
086
087  /**Requires that min and max count both have their default value, 1.
088  */
089  positiveLookbehind("<=",true),
090
091  /**Requires that min and max count both have their default value, 1.
092  */
093  negativeLookbehind("<!",true),
094
095/**
096
097    This value indicates that the given pattern should match independently
098    of any larger pattern in which it may be embedded, thereby avoiding
099    backtracking when you know you want a given subpattern to be matched;
100    it requires that min and max count both have their default value, 1.
101    The only reason to ever use {@link #independent}, is to
102    optimize performance.
103
104*/
105
106  independent("?>",true),
107  ;
108  private final String string;
109  private final boolean oneOneCount; //ether must be or not must be.
110
111  Policy (String string,boolean oneOneCount)
112  {
113     this.string=string;
114     this.oneOneCount=oneOneCount;
115  }
116  String getIntroduction(){return string;}
117
118  /**Return whether or not this Policy == greedy*/
119  public boolean isDefault(){return this==greedy;}
120
121  /** Checks that the min and max count of pattern are compatible with this
122  Policy.
123  @throws BadCountException if pattern's min and max count are not compatible
124  with this Policy.
125  */
126
127  public Policy checkCounts(Pattern pattern)
128  {
129     int min=pattern.getMinCount();
130     Integer max=pattern.getMaxCount();
131     if(oneOneCount)
132     {
133        if(min==1 && max.equals( new Integer(1))) return this;
134     }
135     else
136     {
137        if(min != 1 || ! new Integer(1).equals(max)) return this;
138     }
139     throw new BadCountException(pattern);
140
141  }
142
143  StringBuilder toString(StringBuilder sb,Verbose verbose)
144  {
145     if(sb==null) sb=new StringBuilder();
146     switch(verbose)
147     {
148        case no: return sb;
149        case yes: if(isDefault()) return sb; else break;
150        case very: break;
151     }
152     sb.append("policy=" + this);
153     return sb;
154  }
155
156  public String toString() {return getClass().getSimpleName()+"."+name();}
157
158}
