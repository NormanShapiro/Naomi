033package org.naomi.regex;
034
035/**
036
037  Each constant defined by this Enum specifies a boundary (word, line, etc.)
038  that is to be matched by an instance of {@link BoundaryPattern} instantiated
039  with this value.
040
041 */
042
043public enum Boundary
044{
045  line("^","$"),
046  word("\\b","\\b"),
047  nonWord("\\B","\\B"),
048
049  /** All of input */
050  all("\\A","\\z"),
051
052  /** All of input except for line terminator, if any*/
053  allButForTerminator(null,"\\Z"),
054  ;
055
056  private String left, right;
057
058  private Boundary(String left, String right)
059  {
060     this.left=left;
061     this.right=right;
062  }
063
064  public String getLeft()
065  {
066     if(left==null) throw new IllegalArgumentException();
067      return left;
068  }
069
070  public String getRight()
071  {
072     if(right==null) throw new IllegalArgumentException();
073     return right;
074  }
075  public String toString() {return getClass().getSimpleName()+"."+name();}
076}
